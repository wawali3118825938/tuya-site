// 上传照片 function uploadPhoto() { const author = document.getElementById('author').value; const photographer = document.getElementById('photographer').value; const findTime = document.getElementById('findTime').value; const location = document.getElementById('location').value; const findLocation = document.getElementById('findLocation').value; const status = document.getElementById('status').value; const description = document.getElementById('description').value; const file = fileInput.files[0]; // 验证 if (!author || !photographer || !findTime || !location || !findLocation || !status) { alert('请填写所有带*的信息！'); return; } if (!file) { alert('请先选择照片！'); return; } // 显示加载中 document.getElementById('loading').style.display = 'block'; document.getElementById('photoForm').style.display = 'none'; // 读取照片 const reader = new FileReader(); reader.onload = function(e) { const imageData = e.target.result; // 准备数据 const submissionData = { id: Date.now(), type: 'photo', nickname: user.nickname, email: user.email, imageData: imageData, location: location, author: author, photographer: photographer, find_time: findTime, find_location: findLocation, status: status, description: description || '无', submission_time: new Date().toLocaleString() }; // 保存到待审核列表 const pending = JSON.parse(localStorage.getItem('guangzhou_pending')) || []; pending.push(submissionData); localStorage.setItem('guangzhou_pending', JSON.stringify(pending)); // 发送邮件通知 sendPhotoEmail({ nickname: user.nickname, email: user.email, location: location, author: author, photographer: photographer, find_time: findTime, find_location: findLocation, status: status, description: description || '无', submission_time: new Date().toLocaleString() }); }; reader.readAsDataURL(file); } // 上传文章 function uploadArticle() { const articleType = document.getElementById('articleType').value; const articleTitle = document.getElementById('articleTitle').value; const articleAuthor = document.getElementById('articleAuthor').value; const articleContent = document.getElementById('articleContent').value; // 验证 if (!articleType || !articleTitle || !articleAuthor || !articleContent) { alert('请填写所有带*的信息！'); return; } // 显示加载中 document.getElementById('loading').style.display = 'block'; document.getElementById('articleForm').style.display = 'none'; // 准备数据 const submissionData = { id: Date.now(), type: 'article', nickname: user.nickname, email: user.email, article_type: articleType, // 添加这个字段，用于管理员后台显示板块 title: articleTitle, article_author: articleAuthor, content: articleContent, submission_time: new Date().toLocaleString() }; // 保存到待审核列表 const pending = JSON.parse(localStorage.getItem('guangzhou_pending')) || []; pending.push(submissionData); localStorage.setItem('guangzhou_pending', JSON.stringify(pending)); // 发送邮件通知 sendArticleEmail({ nickname: user.nickname, email: user.email, type: articleType, title: articleTitle, article_author: articleAuthor, content: articleContent.substring(0, 500) + (articleContent.length > 500 ? '...' : ''), submission_time: new Date().toLocaleString() }); }
