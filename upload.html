// 检查登录状态 const userData = localStorage.getItem('guangzhou_user'); if (!userData) { alert('请先登录！'); window.location.href = 'login.html'; } // 显示用户信息 const user = JSON.parse(userData); document.getElementById('userName').textContent = user.nickname; document.getElementById('userAvatar').textContent = user.nickname.charAt(0).toUpperCase(); document.getElementById('photographer').value = user.nickname; document.getElementById('findTime').valueAsDate = new Date(); document.getElementById('photoForm').style.display = 'none'; // 切换选项卡 function showTab(tabName) { // 隐藏所有选项卡内容 document.querySelectorAll('.tab-content').forEach(content => { content.classList.remove('active'); }); document.querySelectorAll('.tab').forEach(tab => { tab.classList.remove('active'); }); // 显示选中的选项卡 if (tabName === 'photo') { document.getElementById('photoTab').classList.add('active'); document.querySelector('.tab:nth-child(1)').classList.add('active'); } else { document.getElementById('articleTab').classList.add('active'); document.querySelector('.tab:nth-child(2)').classList.add('active'); } } // 设置拖拽功能 const uploadBox = document.querySelector('.upload-box'); const fileInput = document.getElementById('fileInput'); uploadBox.addEventListener('click', () => fileInput.click()); fileInput.addEventListener('change', (e) => { if (e.target.files[0]) showPreview(e.target.files[0]); }); uploadBox.addEventListener('dragover', (e) => { e.preventDefault(); uploadBox.style.borderColor = 'var(--primary)'; }); uploadBox.addEventListener('dragleave', () => { uploadBox.style.borderColor = 'rgba(255,255,255,0.1)'; }); uploadBox.addEventListener('drop', (e) => { e.preventDefault(); uploadBox.style.borderColor = 'rgba(255,255,255,0.1)'; if (e.dataTransfer.files[0]) showPreview(e.dataTransfer.files[0]); }); // 预览照片 function showPreview(file) { if (!file.type.startsWith('image/')) { alert('请选择图片文件！'); return; } if (file.size > 10 * 1024 * 1024) { alert('图片太大，请选择小于10MB的图片'); return; } const reader = new FileReader(); reader.onload = function(e) { document.getElementById('previewImage').src = e.target.result; document.getElementById('preview').style.display = 'block'; document.getElementById('photoForm').style.display = 'block'; }; reader.readAsDataURL(file); } // 上传照片 function uploadPhoto() { const author = document.getElementById('author').value; const photographer = document.getElementById('photographer').value; const findTime = document.getElementById('findTime').value; const location = document.getElementById('location').value; const findLocation = document.getElementById('findLocation').value; const status = document.getElementById('status').value; const description = document.getElementById('description').value; const file = fileInput.files[0]; // 验证 if (!author || !photographer || !findTime || !location || !findLocation || !status) { alert('请填写所有带*的信息！'); return; } if (!file) { alert('请先选择照片！'); return; } // 显示加载中 document.getElementById('loading').style.display = 'block'; document.getElementById('photoForm').style.display = 'none'; // 读取照片 const reader = new FileReader(); reader.onload = function(e) { const imageData = e.target.result; // 准备数据 const submissionData = { id: Date.now(), type: 'photo', nickname: user.nickname, email: user.email, imageData: imageData, location: location, author: author, photographer: photographer, find_time: findTime, find_location: findLocation, status: status, description: description || '无', submission_time: new Date().toLocaleString() }; // 保存到待审核列表 const pending = JSON.parse(localStorage.getItem('guangzhou_pending')) || []; pending.push(submissionData); localStorage.setItem('guangzhou_pending', JSON.stringify(pending)); // 发送邮件通知 sendPhotoEmail({ nickname: user.nickname, email: user.email, location: location, author: author, photographer: photographer, find_time: findTime, find_location: findLocation, status: status, description: description || '无', submission_time: new Date().toLocaleString() }); }; reader.readAsDataURL(file); } // 上传文章 function uploadArticle() { const articleType = document.getElementById('articleType').value; const articleTitle = document.getElementById('articleTitle').value; const articleAuthor = document.getElementById('articleAuthor').value; const articleContent = document.getElementById('articleContent').value; // 验证 if (!articleType || !articleTitle || !articleAuthor || !articleContent) { alert('请填写所有带*的信息！'); return; } // 显示加载中 document.getElementById('loading').style.display = 'block'; document.getElementById('articleForm').style.display = 'none'; // 准备数据 const submissionData = { id: Date.now(), type: 'article', nickname: user.nickname, email: user.email, article_type: articleType, title: articleTitle, article_author: articleAuthor, content: articleContent, submission_time: new Date().toLocaleString() }; // 保存到待审核列表 const pending = JSON.parse(localStorage.getItem('guangzhou_pending')) || []; pending.push(submissionData); localStorage.setItem('guangzhou_pending', JSON.stringify(pending)); // 发送邮件通知 sendArticleEmail({ nickname: user.nickname, email: user.email, type: articleType, title: articleTitle, article_author: articleAuthor, content: articleContent.substring(0, 500) + (articleContent.length > 500 ? '...' : ''), submission_time: new Date().toLocaleString() }); } // 发送照片投稿邮件 function sendPhotoEmail(data) { const templateParams = { nickname: data.nickname, email: data.email, location: data.location, author: data.author, photographer: data.photographer, find_time: data.find_time, find_location: data.find_location, status: data.status, description: data.description, submission_time: data.submission_time }; // 已替换为您的 Service ID 和照片模板 ID emailjs.send('service_nte32ag', 'template_udz9tlv', templateParams) .then(function(response) { console.log('SUCCESS!', response.status, response.text); showSuccess('照片提交成功！管理员会尽快审核，审核通过后会邮件通知您。'); }, function(error) { console.log('FAILED...', error); alert('提交失败，请稍后重试。错误：' + JSON.stringify(error)); document.getElementById('loading').style.display = 'none'; document.getElementById('photoForm').style.display = 'block'; }); } // 发送文章投稿邮件 function sendArticleEmail(data) { const templateParams = { nickname: data.nickname, email: data.email, type: data.type, title: data.title, article_author: data.article_author, content: data.content, submission_time: data.submission_time }; // 已替换为您的 Service ID 和文章模板 ID emailjs.send('service_nte32ag', 'template_bwg3ho9', templateParams) .then(function(response) { console.log('SUCCESS!', response.status, response.text); showSuccess('文章投稿成功！管理员会尽快审核，审核通过后会邮件通知您。'); }, function(error) { console.log('FAILED...', error); alert('提交失败，请稍后重试。错误：' + JSON.stringify(error)); document.getElementById('loading').style.display = 'none'; document.getElementById('articleForm').style.display = 'block'; }); } // 显示成功信息 function showSuccess(message) { document.getElementById('loading').style.display = 'none'; document.getElementById('successMessage').textContent = message; document.getElementById('success').style.display = 'block'; }
